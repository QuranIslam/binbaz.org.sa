  5. تفسير قوله تعالى: {يَا أَيُّهَا الَّذِينَ آمَنُوا إِنَّمَا الْخَمْرُ وَالْمَيْسِرُ وَالْأَنْصَابُ وَالْأَزْلَامُ رِجْسٌ مِنْ عَمَلِ الشَّيْطَانِ..}

تفسير قوله تعالى: {يَا أَيُّهَا الَّذِينَ آمَنُوا إِنَّمَا الْخَمْرُ وَالْمَيْسِرُ وَالْأَنْصَابُ وَالْأَزْلَامُ رِجْسٌ مِنْ عَمَلِ الشَّيْطَانِ..}

 
  •  
تحميل المادة

حديثٌ آخر: قال الحافظ أبو بكر البيهقي: أنبأنا أبو الحسين ابن بُشران: أنبأنا إسماعيل بن محمد الصَّفار: حدثنا محمد بن عبيدالله المنادي: حدثنا وهب بن جرير: حدثنا شعبة، عن سماك، عن مصعب بن سعد، عن سعدٍ، قال: أُنزلت في الخمر أربع آيات. فذكر الحديث، قال: وصنع رجلٌ من الأنصار طعامًا فدعانا، فشربنا الخمر قبل أن تُحرم حتى انتشينا؛ فتفاخرنا، فقالت الأنصار: نحن أفضل. وقالت قريش: نحن أفضل. فأخذ رجلٌ من الأنصار لحى جزورٍ فضرب به أنف سعدٍ ففزره، وكانت أنفُ سعدٍ مفزورة، فنزلت: إِنَّمَا الْخَمْرُ وَالْمَيْسِرُ إلى قوله تعالى: فَهَلْ أَنْتُمْ مُنْتَهُونَ [المائدة:90-91]. أخرجه مسلمٌ من حديث شُعبة.

حديثٌ آخر: قال البيهقي: وأخبرنا أبو نصر ابن قتادة: أنبأنا أبو علي الرفاء: حدثنا علي بن عبدالعزيز: حدثنا حجاج بن منهال: حدثنا ربيعة بن كلثوم: حدثني أبي، عن سعيد بن جبير، عن ابن عباسٍ، قال: إنما نزل تحريمُ الخمر في قبيلتين من قبائل الأنصار شربوا، فلمَّا أن ثمل القومُ عبث بعضُهم ببعضٍ، فلمَّا أن صحوا جعل الرجلُ يرى الأثر بوجهه ورأسه ولحيته، فيقول: صنع بي هذا أخي فلان. وكانوا إخوةً، ليس في قلوبهم ضغائن، فيقول: والله لو كان بي رؤوفًا رحيمًا ما صنع بي هذا. حتى وقعت الضَّغائن في قلوبهم، فأنزل الله تعالى هذه الآية: يَا أَيُّهَا الَّذِينَ آمَنُوا إِنَّمَا الْخَمْرُ وَالْمَيْسِرُ وَالْأَنْصَابُ وَالْأَزْلَامُ رِجْسٌ مِنْ عَمَلِ الشَّيْطَانِ إلى قوله تعالى: فَهَلْ أَنْتُمْ مُنْتَهُونَ، فقال أناسٌ من المتكلفين: هي رجسٌ، وهي في بطن فلان، وقد قُتل يوم أحد! فأنزل الله تعالى: لَيْسَ عَلَى الَّذِينَ آمَنُوا وَعَمِلُوا الصَّالِحَاتِ جُنَاحٌ فِيمَا طَعِمُوا [المائدة:93] إلى آخر الآية.

ورواه النَّسائي في "التفسير" عن محمد بن عبدالرحيم (صاعقة)، عن حجاج بن منهال.

حديثٌ آخر: قال ابن جرير: حدثني محمد بن خلف: حدثنا سعيد بن محمد الجرمي، عن أبي تميلة، عن سلام -مولى حفص أبي القاسم-، عن ابن بُريدة، عن أبيه، قال: بينا نحن قعودٌ على شرابٍ لنا، ونحن على رملةٍ، ونحن ثلاثة أو أربعة، وعندنا باطية لنا، ونحن نشرب الخمر حلًّا، إذ قمتُ حتى آتي رسول الله ﷺ فأُسلم عليه، إذ نزل تحريمُ الخمر: يَا أَيُّهَا الَّذِينَ آمَنُوا إِنَّمَا الْخَمْرُ وَالْمَيْسِرُ إلى آخر الآيتين: فَهَلْ أَنْتُمْ مُنْتَهُونَ، فجئتُ إلى أصحابي فقرأتها عليهم إلى قوله: فَهَلْ أَنْتُمْ مُنْتَهُونَ، قال: وبعض القوم شربته في يده، قد شرب بعضها، وبقي بعضٌ في الإناء، فقال بالإناء تحت شفته العليا كما يفعل الحجّام، ثم صبّوا ما في باطيتهم، فقالوا: انتهينا ربّنا.

الشيخ: الله أكبر، .

س: ما صحّة هذا السند؟

ج: محلّ نظرٍ، الحمد لله، الأحاديث كافية، الآية الكريمة والأحاديث الصَّحيحة كافية.

س: قوله أنَّ أنف سعدٍ كانت مفزورةً؟

ج: بسبب الضَّربة التي ضربه إياها.

س: فزره، وكانت مفزورة؟

ج: مهزورًا، مهزورًا بسبب الضَّربة.

حديثٌ آخر: قال البخاري: حدَّثنا صدقة بن الفضل: أخبرنا ابنُ عيينة، عن عمرو، عن جابرٍ قال: صبح أناسٌ غداة أحدٍ الخمر، فقُتلوا من يومهم جميعًا شُهداء، وذلك قبل تحريمها. هكذا رواه البخاري في "تفسيره" من "صحيحه".

وقد رواه الحافظ أبو بكر البزار في "مسنده": حدثنا أحمد بن عبدة: حدثنا سفيان، عن عمرو بن دينار، سمع جابر بن عبدالله يقول: اصطبح ناسٌ الخمر من أصحاب النبي ﷺ، ثم قُتلوا شُهداء يوم أحد، فقالت اليهود: فقد مات بعضُ الذين قُتلوا وهي في بطونهم! فأنزل الله: لَيْسَ عَلَى الَّذِينَ آمَنُوا وَعَمِلُوا الصَّالِحَاتِ جُنَاحٌ فِيمَا طَعِمُوا. ثم قال: وهذا إسنادٌ صحيحٌ. وهو كما قال، ولكن في سياقه غرابة.

الشيخ: وهو من جهة كلام اليهود.

س: الغرابة من أين؟

ج: قد يكون من جهة كلام اليهود، نعم.

حديثٌ آخر: قال أبو داود الطّيالسي: حدثنا شعبة، عن أبي إسحاق، عن البراء بن عازب، قال: لما نزل تحريمُ الخمر قالوا: كيف بمَن كان يشربها قبل أن تحرم؟! فنزلت: لَيْسَ عَلَى الَّذِينَ آمَنُوا وَعَمِلُوا الصَّالِحَاتِ جُنَاحٌ فِيمَا طَعِمُوا الآية. ورواه الترمذي عن بندار، غندر.

الشيخ: عن بندار، عن غندر. بندار: محمد بن بشار، وغندر: محمد بن جعفر، شخصان.

عن بندار، عن غندر، عن شعبة، به نحوه، وقال: حسنٌ صحيحٌ.

حديثٌ آخر: قال الحافظ أبو يعلى الموصلي: حدثنا جعفر بن حميد الكوفي: حدثنا يعقوب القمّي، عن عيسى بن جارية، عن جابر بن عبدالله، قال: كان رجلٌ يحمل الخمر من خيبر إلى المدينة، فيبيعها من المسلمين، فحمل منها بمالٍ، فقدم بها المدينة، فلقيه رجلٌ من المسلمين فقال: يا فلان، إنَّ الخمر قد حُرّمت. فوضعها حيث انتهى على تلٍّ، وسجى عليها بأكسيةٍ، ثم أتى النبي ﷺ فقال: يا رسول الله، بلغني أنَّ الخمر قد حُرّمت؟ قال: أجل، قال: لي أن أردّها على مَن ابتعتُها منه؟ قال: لا يصلح ردّها، قال: لي أن أهديها إلى مَن يُكافئني منها؟ قال: لا، قال: فإنَّ فيها مالًا ليتامى في حجري. قال: إذا أتانا مالُ البحرين فأتنا نُعوّض أيتامك من مالهم، ثم نادى بالمدينة، فقال رجلٌ: يا رسول الله، الأوعية ننتفع بها؟ قال: فحلّوا أوكيتها، فانصبّت حتى استقرّت في بطن الوادي. هذا حديثٌ غريبٌ.

الشيخ: انظر: جعفر بن حميد في "التقريب".

الطالب: جعفر بن حميد القرشي، أو العبسي –بموحدة-، أبو محمد، الكوفي، عن عبيدالله بن إياد، وعنه مسلم فرد حديثٍ مقرونًا بيحيى بن يحيى، وثَّقه البستي، وقال مطين: توفي سنة 240.

الشيخ: الأخير: عيسى بن جارية؟

قارئ المتن: الثاني: يعقوب القمّي.

الشيخ: معروف، بعده؟

قارئ المتن: عيسى بن جارية.

الطالب: هذا جعفر يا شيخ، جعفر بن حميد العبسي، الكوفي، أبو محمد، المعروف بزمبقة، ثقة، من العاشرة، مات سنة أربعين. (مسلم).

الشيخ: انظر: "عيسى بن جارية".

المقصود في هذا الدّلالة على أنَّ الذي لم يعرف تحريم الخمر فلا تُشقّ الأواني، يُريقها وينتفع بأوانيه، أمَّا الذي تعمّد بيع الخمر بعدما حرمت فهذا يُعاقب بشقّها.

الطالب: عيسى بن جارية –بالجيم- الأنصاري، المدني، فيه لين، من الرابعة. (ابن ماجه).

الشيخ: غيره ثانٍ؟

الطالب: ما في يا شيخ.

الشيخ: وعندك؟

الطالب: عيسى بن جارية –بالجيم- الأنصاري، المدني، عن جابر، وعنه زيد ابن أبي أنيسة، وثقه ابن حبان، وقال أبو داود: منكر الحديث.

الشيخ: نعم، اللهم اغفر لنا وله.

حديثٌ آخر: قال الإمام أحمد: حدثنا وكيع: حدثنا سفيان، عن السدي، عن أبي هبيرة -وهو يحيى بن عباد الأنصاري-، عن أنس بن مالك: أنَّ أبا طلحة سأل رسول الله ﷺ عن أيتامٍ في حجره ورثوا خمرًا، فقال: أهرقها، قال: أفلا نجعلها خلًّا؟ قال: لا.

ورواه مسلم وأبو داود والترمذي من حديث الثوري، به نحوه.

حديثٌ آخر: قال ابنُ أبي حاتم: حدثنا عبدالله بن رجاء: حدثنا عبدالعزيز بن سلمة: حدثنا هلال ابن أبي هلال، عن عطاء بن يسار، عن عبدالله بن عمرو، قال: إنَّ هذه الآية التي في القرآن: يَا أَيُّهَا الَّذِينَ آمَنُوا إِنَّمَا الْخَمْرُ وَالْمَيْسِرُ وَالْأَنْصَابُ وَالْأَزْلَامُ رِجْسٌ مِنْ عَمَلِ الشَّيْطَانِ فَاجْتَنِبُوهُ لَعَلَّكُمْ تُفْلِحُونَ [المائدة:90]. قال: هي في التوراة: إنَّ الله أنزل الحقَّ ليُذهب به الباطل، ويُبطل به اللَّعب والمزامير، والزّفن والكبارات. يعني: البرابط والزّمارات، يعني به الدّف والطَّنابير والشّعر، والخمر مُرّة لمن طعمها، أقسم اللهُ بيمينه وعزّته مَن شربها بعدما حرّمتها لأعطشنّه يوم القيامة، ومَن تركها بعدما حرّمتها لأسقينَّه إياها في حظيرة القدس. وهذا إسنادٌ صحيحٌ.

الشيخ: وفيما أنزل الله في كتابه العظيم الكفاية، والحمد لله.

س: كانت مُحرّمةً على مَن قبلنا؟

ج: هذا يرويه عبدالله بن عمرو عن التوراة، والله أعلم.

أقسم الله بيمينه وعزّته ..... مَن شربها بعدما حرّمتها لأعطشنّه يوم القيامة، ومَن تركها بعدما حرّمتها لأسقينّه إياها في حظيرة القدس. وهذا إسنادٌ صحيحٌ.

الشيخ: يعني: صحيحًا إلى عبدالله بن عمرو، وهو يروي عن الإسرائيليات كثيرًا، عبدالله بن عمرو يروي عن بني إسرائيل كثيرًا -رحمه الله.

س: هل يُشترط مع الغناء آلة مزمارٍ؟

ج: لا، ولو ما معه مزمار، إذ الغناء مذمومٌ، يُمنع ولو ما معه مزمار.

س: ما هو الزّفن يا شيخ؟

ج: الظاهر لعب الزنوج وغناؤهم ورقصهم.

س: مَن باعها بعد تحريمها تُكسّر الأواني؟

ج: إيه، يستحقّ، يُعاقَب، نعم.

س: الأناشيد الإسلامية؟

ج: الشعر العربي الذي لا حرجَ فيه، سُمّي: أناشيد، أو غيره، لا حرج: إنَّ من الشعر حكمةً.

س: الأناشيد يرفعونها بصوتٍ جماعي؟

ج: على كل حالٍ، هذا ما هو بقراءة، هذا إن كان شعرًا طيبًا، شعرًا سليمًا ما يضرّ، نعم.

س: وإذا كانت فيها فتنة، فيها تمطيط وترقيق؟

ج: على كلٍّ ينبغي أن يُلقى الشعر الطّيب بصوتٍ ما فيه فتنة.

س: الفرق بين القصيد والشّعر؟

ج: القصيد هو الشعر، إذا كان الشعرُ سليمًا ما فيه ..... يُسمّى: قصيدة، ويُسمّى: شعرًا.

حديثٌ آخر: قال عبدالله بن وهب: أخبرني عمرو بن الحارث: أنَّ عمرو بن شُعيب حدَّثهم عن أبيه، عن عبدالله بن عمرو بن العاص، عن رسول الله ﷺ قال: مَن ترك الصلاة سكرًا مرةً واحدةً فكأنما كانت له الدنيا وما عليها فسُلبها، ومَن ترك الصلاةَ سكرًا أربع مرات كان حقًّا على الله أن يسقيه من طينة الخبال، قيل: وما طينة الخبال؟ قال: عصارة أهل جهنم. ورواه أحمد من طريق عمرو بن شُعيب.

حديثٌ آخر: قال أبو داود: حدثنا محمد بن رافع: حدثنا إبراهيم بن عمر الصّنعاني، قال: سمعتُ النُّعمان -هو ابن أبي شيبة الجندي- يقول: عن طاوس، عن ابن عباسٍ، عن النبي ﷺ- قال: كل مخمر خمر، وكل مُسكر حرام، ومَن شرب مُسكرًا بخست صلاته أربعين صباحًا، فإن تاب تاب اللهُ عليه، فإن عاد الرابعة كان حقًّا على الله أن يسقيه من طينة الخبال، قيل: وما طينة الخبال يا رسول الله؟ قال: صديد أهل النار، ومَن سقاه صغيرًا لا يعرف حلاله من حرامه كان حقًّا على الله أن يسقيه من طينة الخبال. تفرد به أبو داود.

الشيخ: نسأل الله العافية، نعم.

س: كل مخمر خمر عفا الله عنك، كذا؟

ج: لعله: "مُخَمر"، يعني: أُرصد وحُبس حتى تخمّر.

س: السّند هذا؟

ج: حدثنا أيش؟

قال أبو داود: حدثنا محمد بن رافع: حدثنا إبراهيم بن عمر الصّنعاني، قال: سمعتُ النعمان -هو ابن أبي شيبة الجندي- يقول: عن طاوس، عن ابن عباسٍ.

الشيخ: انظر: النعمان ابن أبي شيبة الجندي.

لكن معناه صحيح في أحاديث أخرى، نسأل الله العافية، نعم، من باب الوعيد، نسأل الله العافية، مثلما جاء فيمَن أتى عرَّافًا فسأله عن شيءٍ لم تُقبل له صلاة أربعين ليلة، نسأل الله العافية.

س: تحديد الأربعين يومًا له وجهٌ؟

ج: الله أحكم وأعلم -جلَّ وعلا-.

س: حديث قتله في الرابعة؟

ج: عند الجمهور منسوخ، وبعض أهل العلم يرى قتله في الرابعة، وإلا فأكثرهم يرون أنَّه منسوخ.

الطالب: النّعمان ابن أبي شيبة عبيد الصّنعاني، أو الجندي -بفتح الجيم والنون-، ثقة، من السابعة. (د).

الشيخ: نعم، طيب.

س: صحيحٌ إسناد الحديث؟

ج: لا بأس به، نعم، نسأل الله العافية.

الله يغفر له، أطال في المقام -رحمه الله-، مع أنَّ الآية -بحمد الله- كافية، جزاه الله خيرًا، نعم.

س: مَن داعب أهله قبل التَّحلل الأول للإحرام، ثم أنزل على إثر هذه المداعبة، ماذا عليه؟

ج: ذبيحة، يذبح ذبيحةً إن كان ما جامع بعد التَّحلل الأول .....، أمَّا الجماع فهو الذي يُفسد، إن كان قبل التَّحلل الأول عليه ذبيحة، وقال بعضهم: فدية الأذى: إطعام ستة مساكين، أو صوم ثلاثة أيام، أو ذبح شاة إن كان بعد التَّحلل الأول.

س: إذا ما استطاع أن يذبح شاةً؟

ج: يصوم ثلاثة أيام، أو يُطعم ستة مساكين، نعم، كفدية الأذى.

س: مَن صلّى خلف إمامٍ يستعجل في صلاته، فلا يتمكن من قراءة الفاتحة، فهل ينفصل عنه؟

ج: بعض الناس يتمطط، وإلا ما هو بمُصلٍّ خلف إمامٍ ما يُكمل الفاتحة، لا حولَ ولا قوةَ إلا بالله.

س: ليس إمامًا أصليًّا، ولكنه ممن يقضون الصلاة بعد الجماعة؟

ج: يمكن المأموم هو الذي تأخّر قليلًا، وتسقط عنه، المأموم تسقط عنه، إذا ما أمكنه سقطت عنه، كالذي جاء والإمام راكع.

س: لكنه في أكثر الركعات لا يتمكّن من قراءة الفاتحة .....؟

ج: إذا صحَّ هذا ما تصحّ صلاته، ولا مَن خلفه، إذا صحَّ هذا فصلاته باطلة، إذا صحَّ هذا: أنَّه ما يطمئن فصلاته باطلة، لكن بعض الناس قد تكون عنده وسوسة، يقول: ما هو بمطمئن، وهو مطمئن، لكن إذا صحَّ أنَّه ما يطمئن فالذي ما يطمئن ما تصحّ صلاته.

س: المأموم لو نسي واجبًا من واجبات الصّلاة فهل له أن يسجد السّهو بعدما يُسلم الإمام؟

ج: لا، ما يسجد وحده إلا إذا كان مسبوقًا، يتحمّل عنه الإمام، نعم.

س: الإهلال بالتَّلبية في الطائرة: إذا كان الرجلُ أو الرجلان في الطَّائرة، ومعهم ركاب عاديون، ليسوا محرمين، فيرفع صوته، فيُنكرون عليه، بعضهم يقول: إنَّك تُزعج مَن معك؟

ج: لا، رفعًا خفيفًا، لا يُؤذي به أحدًا.

س: ...........؟

ج: الأفضل أن يتمتع، هذا هو الأفضل، الذي ينبغي أن يتمتع؛ لأنَّ الميتة عليها عُمرة وحجّ، فيتمتع بعمرةٍ وحجٍّ، وهذا الذي أمر به النبي ﷺ الصَّحابة.

س: ...........؟

ج: ما يُخالف، إلا أن يشرط عليه شيئًا، وإلا لا حرج .....، الحمد لله.
