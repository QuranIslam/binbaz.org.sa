  5. تفسير قوله تعالى: {يَا أَيُّهَا الَّذِينَ آمَنُوا لَا تَتَّخِذُوا الَّذِينَ اتَّخَذُوا دِينَكُمْ هُزُوًا وَلَعِبًا..}

تفسير قوله تعالى: {يَا أَيُّهَا الَّذِينَ آمَنُوا لَا تَتَّخِذُوا الَّذِينَ اتَّخَذُوا دِينَكُمْ هُزُوًا وَلَعِبًا..}

 
  •  
تحميل المادة

يَا أَيُّهَا الَّذِينَ آمَنُوا لَا تَتَّخِذُوا الَّذِينَ اتَّخَذُوا دِينَكُمْ هُزُوًا وَلَعِبًا مِنَ الَّذِينَ أُوتُوا الْكِتَابَ مِنْ قَبْلِكُمْ وَالْكُفَّارَ أَوْلِيَاءَ وَاتَّقُوا اللَّهَ إِنْ كُنْتُمْ مُؤْمِنِينَ ۝ وَإِذَا نَادَيْتُمْ إِلَى الصَّلَاةِ اتَّخَذُوهَا هُزُوًا وَلَعِبًا ذَلِكَ بِأَنَّهُمْ قَوْمٌ لَا يَعْقِلُونَ [المائدة:57-58].

هذا تنفيرٌ من مُوالاة أعداء الإسلام وأهله من الكتابيين والمشركين الذين يتخذون أفضل ما يعمله العاملون، وهي شرائع الإسلام المطهرة، المحكمة، المشتملة على كل خيرٍ دنيوي وأُخروي، يتّخذونها هزوًا، يستهزئون بها، ولعبًا، يعتقدون أنها نوعٌ من اللعب في نظرهم الفاسد، وفكرهم البارد، كما قال القائل:

وكم من عائبٍ قولًا صحيحًا وآفته من الفهم السَّقيم

الشيخ: وهؤلاء عندهم العقيدة الباطلة: عندهم الكفر بالله، والكفر بآياته، والكفر بأنبيائه؛ ولهذا استهزؤوا، اتخذوا آيات الله هزوًا -والعياذ بالله-، ما عندهم إيمانٌ، وبعضهم مُعاند، يفهم ويعرف –كاليهود- ولكنَّه العناد والحسد والبغي، نسأل الله العافية: يَعْرِفُونَهُ كَمَا يَعْرِفُونَ أَبْنَاءَهُمْ [البقرة:146]، ولكن عندهم الحسد والبغي والعدوان.

وأما المنافقون فهم بين كذا وكذا: بين ملبوسٍ عليه، وبين مُعاندٍ، نسأل الله العافية.

وهكذا كفَّار قريش وغيرهم: منهم مَن هو مُعاند يعرف الحقَّ، ومنهم مَن هو جاهل جهلًا مُركبًا يتبع غيره.

فالله عابهم بما عملوا من اتِّخاذ آيات الله هزوًا ولعبًا، وقال في اليهود وأشباههم: وَجَحَدُوا بِهَا وَاسْتَيْقَنَتْهَا أَنْفُسُهُمْ ظُلْمًا وَعُلُوًّا [النمل:14]، وقال تعالى: قَدْ نَعْلَمُ إِنَّهُ لَيَحْزُنُكَ الَّذِي يَقُولُونَ فَإِنَّهُمْ لَا يُكَذِّبُونَكَ وَلَكِنَّ الظَّالِمِينَ بِآيَاتِ اللَّهِ يَجْحَدُونَ [الأنعام:33]، فأبو طالب وغيره من صناديد قريش يعرفون صدق محمدٍ ﷺ، ولكنه العناد والبغي والحسد والتَّكبر، نسأل الله العافية.

وقوله تعالى: مِنَ الَّذِينَ أُوتُوا الْكِتَابَ مِنْ قَبْلِكُمْ وَالْكُفَّارَ "من" هاهنا لبيان الجنس، كقوله: فَاجْتَنِبُوا الرِّجْسَ مِنَ الْأَوْثَانِ [الحج:30].

وقرأ بعضُهم: والكفَّارِ بالخفض عطفًا، وقرأ آخرون بالنَّصب على أنَّه معمول: لَا تَتَّخِذُوا الَّذِينَ اتَّخَذُوا دِينَكُمْ هُزُوًا وَلَعِبًا مِنَ الَّذِينَ أُوتُوا الْكِتَابَ مِنْ قَبْلِكُمْ تقديره: ولا الكفَّار أولياء، أي: لا تتخذوا هؤلاء ولا هؤلاء أولياء.

والمراد بالكفَّار هاهنا المشركون، وكذلك وقع في قراءة ابن مسعودٍ فيما رواه ابن جرير: "لا تتخذوا الذين اتخذوا دينكم هزوًا ولعبًا من الذين أوتوا الكتاب من قبلكم ومن الذين أشركوا".

وقوله: وَاتَّقُوا اللَّهَ إِنْ كُنْتُمْ مُؤْمِنِينَ أي: اتَّقوا الله أن تتخذوا هؤلاء الأعداء لكم ولدينكم أولياء إن كنتم مؤمنين بشرع الله الذي اتّخذه هؤلاء هزوًا ولعبًا، كما قال تعالى: لَا يَتَّخِذِ الْمُؤْمِنُونَ الْكَافِرِينَ أَوْلِيَاءَ مِنْ دُونِ الْمُؤْمِنِينَ وَمَنْ يَفْعَلْ ذَلِكَ فَلَيْسَ مِنَ اللَّهِ فِي شَيْءٍ إِلَّا أَنْ تَتَّقُوا مِنْهُمْ تُقَاةً وَيُحَذِّرُكُمُ اللَّهُ نَفْسَهُ وَإِلَى اللَّهِ الْمَصِيرُ [آل عمران:28].

وقوله: وَإِذَا نَادَيْتُمْ إِلَى الصَّلَاةِ اتَّخَذُوهَا هُزُوًا وَلَعِبًا أي: وكذلك إذا أذنتم داعين إلى الصلاة التي هي أفضل الأعمال لمن يعقل ويعلم من ذوي الألباب اتخذوها أيضًا: هُزُوًا وَلَعِبًا ذَلِكَ بِأَنَّهُمْ قَوْمٌ لَا يَعْقِلُونَ معاني عبادة الله وشرائعه.

وهذه صفات أتباع الشيطان الذي إذا سمع الأذان أدبر وله حصاص، أي: ضراط، حتى لا يسمع التَّأذين، فإذا قُضي التَّأذين أقبل، فإذا ثوب للصلاة أدبر، فإذا قُضي التثويب أقبل حتى يخطر بين المرء وقلبه، فيقول: اذكر كذا، اذكر كذا، لما لم يكن يذكر، حتى يظل الرجل لا يدري كم صلَّى؟ فإذا وجد أحدُكم ذلك فليسجد سجدتين قبل السلام متفق عليه.

وقال الزهري: قد ذكر الله التَّأذين في كتابه فقال: وَإِذَا نَادَيْتُمْ إِلَى الصَّلَاةِ اتَّخَذُوهَا هُزُوًا وَلَعِبًا ذَلِكَ بِأَنَّهُمْ قَوْمٌ لَا يَعْقِلُونَ. رواه ابن أبي حاتم.

وقال أسباط: عن السدي في قوله: وَإِذَا نَادَيْتُمْ إِلَى الصَّلَاةِ اتَّخَذُوهَا هُزُوًا وَلَعِبًا قال: كان رجلٌ من النصارى بالمدينة إذا سمع المنادي يُنادي: "أشهد أنَّ محمدًا رسول الله" قال: حرق الكاذب. فدخلت خادمةٌ ليلةً من الليالي بنارٍ وهو نائم، وأهله نيام، فسقطت شرارةٌ فأحرقت البيت، فاحترق هو وأهله. رواه ابن جرير، وابن أبي حاتم.

الشيخ: لو صحَّ معناه: أنَّ الله عجّل له ما قال، هو كذَّاب وكاذب فحرق.

س: دعا على نفسه؟

ج: على المعنى دعا على نفسه، فهو الكاذب، والذي قال: "أشهد أنَّ محمدًا رسول الله" الصادق، فإن صحّت القصة فالمعنى: أنَّ الله عجَّل له العقوبة كما دعا على نفسه.

س: ذكر ابن جرير في الآية: ولا تتّخذوا الذين اتّخذوا دينكم هزوًا ولعبًا من الذين أوتوا الكتاب من قبلكم والذين أشركوا؟

ج: الَّذِينَ أُوتُوا الْكِتَابَ مِنْ قَبْلِكُمْ وَالْكُفَّارَ أَوْلِيَاءَ هكذا الآية.

س: عفا الله عنك، في اختلاف؟

ج: صلحها.

س: لكن قراءة ابن مسعود توضيح؟

ج: وأيش عندك؟

س: والمشركين.

ج: أيش عندك؟ عن ابن مسعودٍ أيش؟

قارئ المتن: والمراد بالكفَّار هاهنا المشركون، وكذا وقع في رواية ابن مسعود فيما رواه ابن جرير: لا تتخذوا الذين اتخذوا دينكم هزوًا ولعبًا من الذين أوتوا الكتاب من قبلكم ومن الذين أشركوا.

الشيخ: يعني: تفسير معنى الكفَّار، والكفار هم عباد الأوثان، أُوتُوا الْكِتَابَ هؤلاء اليهود والنَّصارى، وبعدهم الكفار، يعني: عباد الأوثان، نسأل الله العافية.

س: لكن هذه القراءة قراءة تفسيرية، أو قراءة مُتعمدة؟

ج: ينظر في صحتها، إن صحَّت فمعناها تفسير، يفسر معنى الكفار، وأنهم المشركون، وقد تكون أيضًا قراءة من الحروف السبعة؛ لأنَّ الحروف السبعة معناها قراءة متّحدة في المعنى، وإن اختلف اللَّفظ في الأحرف السبعة.

وذكر محمد بن إسحاق بن يسار في "السيرة": أنَّ رسول الله ﷺ دخل الكعبة عام الفتح ومعه بلال، فأمره أن يُؤذن، وأبو سفيان بن حرب وعتاب بن أسيد والحارث بن هشام جلوس بفناء الكعبة، فقال عتاب بن أسيد: لقد أكرم اللهُ أسيدًا أن لا يكون سمع هذا فيسمع منه ما يغيظه. وقال الحارث بن هشام: أما والله لو أعلم أنَّه محقٌّ لاتّبعته. فقال أبو سفيان: لا أقول شيئًا، لو تكلّمتُ لأخبرت عني هذه الحصى. فخرج عليهم النبي ﷺ فقال: قد علمتُ الذي قلتم، ثم ذكر ذلك لهم، فقال الحارث وعتاب: نشهد أنَّك رسول الله، ما اطّلع على هذا أحدٌ كان معنا فنقول: أخبرك.

الشيخ: وعتاب هذا أسلم، كلّهم أسلموا وحسن إسلامهم: الحارث بن هشام وأبو سفيان، وعتاب جعله النبي ﷺ أميرًا على مكة أيام الفتح: عتاب بن أسيد ابن أبي العيص ابن أمية، وكان شابًّا في سنِّ الحادية والعشرين حين أمره النبي على أهل مكّة.

وقال الإمام أحمد: حدثنا روح بن عبادة: حدثنا ابنُ جريج: أخبرنا عبدالعزيز بن عبدالملك ابن أبي محذورة: أنَّ عبدالله بن محيريز أخبره، وكان يتيمًا في حجر أبي محذورة، قال: قلتُ لأبي محذورة: يا عم، إني خارجٌ إلى الشام، وأخشى أن أُسأل عن تأذينك. فأخبرني أنَّ أبا محذورة قال له: نعم، خرجتُ في نفرٍ، وكنا في بعض طريق حنين مقفل رسول الله ﷺ من حنين، فلقينا رسولُ الله ﷺ ببعض الطريق، فأذّن مُؤذّن رسول الله ﷺ بالصلاة عند رسول الله ﷺ، فسمعنا صوتَ المؤذن ونحن مُتنكبون، فصرخنا نحكيه ونستهزئ به، فسمع رسولُ الله ﷺ، فأرسل إلينا إلى أن وقفنا بين يديه، فقال رسولُ الله ﷺ: أيّكم الذي سمعتُ صوتَه قد ارتفع؟ فأشار القومُ كلّهم إليَّ، وصدقوا، فأرسل كلّهم وحبسني، وقال: قم فأذّن، فقمتُ ولا شيء أكره إليَّ من رسول الله ﷺ، ولا مما يأمرني به، فقمتُ بين يدي رسول الله ﷺ، فألقى عليَّ رسولُ الله ﷺ التَّأذين هو بنفسه، قال: قل: الله أكبر الله أكبر، أشهد أن لا إله إلا الله، أشهد أن لا إله إلا الله، أشهد أنَّ محمدًا رسول الله، أشهد أنَّ محمدًا رسول الله، ثم قال لي: ارجع فمدد من صوتك، ثم قال: أشهد أن لا إله إلا الله، أشهد أن لا إله إلا الله، أشهد أنَّ محمدًا رسول الله، أشهد أنَّ محمدًا رسول الله، حيَّ على الصلاة، حيَّ على الصلاة، حيَّ على الفلاح، حيَّ على الفلاح، الله أكبر الله أكبر، لا إله إلا الله، ثم دعاني حين قضيتُ التَّأذين فأعطاني صرَّةً فيها شيء من فضةٍ، ثم وضع يدَه على ناصية أبي محذورة، ثم أمرها على وجهه، ثم بين ثدييه، ثم على كبده، حتى بلغت يدُ رسول الله ﷺ سرةَ أبي محذورة، ثم قال رسولُ الله ﷺ: بارك الله فيك، وبارك عليك، فقلتُ: يا رسول الله، مُرني بالتَّأذين بمكّة. فقال: قد أمرتُك به، وذهب كل شيءٍ كان لرسول الله ﷺ من كراهةٍ، وعاد ذلك كلّه محبةً لرسول الله ﷺ.

فقدمتُ على عتاب بن أسيد عامل رسول الله ﷺ بمكة فأذّنت معه بالصلاة عن أمر رسول الله ﷺ.

وأخبرني ذلك مَن أدركتُ من أهلي ممن أدرك أبا محذورة على نحو ما أخبرني عبدالله بن محيريز. هكذا رواه الإمام أحمد.

وقد أخرجه مسلم في "صحيحه"، وأهل السنن الأربعة من طريقٍ عن عبدالله بن محيريز، عن أبي محذورة، واسمه: سمرة بن معير بن لوذان، أحد مُؤذني رسول الله ﷺ الأربعة، وهو مُؤذن أهل مكّة، وامتدت أيامه رضي الله عنه وأرضاه.

الشيخ: وهذا هو الأذان الذي فيه الترجيع، وهو أن يأتي بالشَّهادتين بصوتٍ ليس بالرَّفيع، ثم يرفع بهما صوته مرةً أخرى، يُقال لها: ترجيع؛ لأنَّه أتى بهما، ثم رجع إليهما، هذا أحد نوعي الأذان.

والنوع الثاني: عدم الترجيع، وهو الذي كان يُؤذن به بلال بين يدي النبي ﷺ في المدينة، وهو الذي عليه العمل الآن هنا: عدم الترجيع.

وفي رواية أهل السنن: أنَّ التكبير أربع في أول الأذان، كما هو ثابت في الأحاديث الأخرى: أربع في أوله، واثنتان في آخره، والبقية اثنتان اثنتان، ما عدا كلمة التوحيد، وهي واحدة في الأذان والإقامة.

س: معنى الترجيع؟

ج: الترجيع يعني: يُكرر الشهادتين، الأول بصوتٍ ليس بالرفيع، والثاني بصوتٍ أرفع، هذا معنى الترجيع: أشهد أن لا إله إلا الله، أشهد أن لا إله إلا الله، أشهد أنَّ محمدًا رسول الله، أشهد أنَّ محمدًا رسول الله، ثم يرجع فيقول: أشهد، ويرفع صوته زيادةً.

س: يرجع إلى شهادة أن لا إله إلا الله؟

ج: نعم، يأتي بالشَّهادتين جميعًا كلتاهما.

وفيما جرى علمٌ من أعلام النبوة في هذا وأمثاله؛ فإنه ﷺ لما مسح على ناصية أبي محذورة وعلى صدره أذهب اللهُ ما في قلبه من الشَّر، وصار بدل ذلك المحبّة والتَّصديق والهداية، ثم لزم الأذان كما علّمه النبي ﷺ، وهكذا ما جرى لأبي سفيان والحارث وعتاب من الهداية والتوفيق، كل هذا من فضل الله جلَّ وعلا ورحمته ، بعدما كانوا يتّخذون آيات الله هزوًا هداهم الله، وأصلح قلوبهم، وأدخل عليهم الإيمان، وهو جلَّ وعلا ذو الفضل والرحمة: قُلْ بِفَضْلِ اللَّهِ وَبِرَحْمَتِهِ فَبِذَلِكَ فَلْيَفْرَحُوا هُوَ خَيْرٌ مِمَّا يَجْمَعُونَ [يونس:58]، هو الذي يهدي مَن يشاء .

س: أحسن الله إليك، مَن قال: إنَّ أذان أبي محذورة الآن لا يُشرع، ولا ينبغي؛ لأنَّ فيه تشويشًا على العامَّة؟

ج: كلّه سنة، الترجيع سنة، فمَن فعله فلا بأس، ولكن الإنسان يُؤذّن كما يُؤذّن الناسُ الآن؛ لعدم التَّشويش، تكون طريقة واحدة أولى، كما كانت الطريقةُ واحدةً في عهد النبي ﷺ في المدينة، كان في الأذان بلال.

س: هل له حكمة مُعينة هذا الترجيع؟

ج: لعلَّ الحكمة: ترسيخ التوحيد في قلب أبي محذورة؛ لأنَّه حديث العهد بالإسلام، إذا كرر ذلك كان أثبت للإيمان في قلبه.

س: ...............؟

ج: المقصود أنَّ أذان بلال أفضل، كما كان بين يدي النبي ﷺ، وعليه العمل الآن.

س: مَن أذّن أذان بلال، وأقام إقامة أبي محذورة؟

ج: يُؤذن بأذان بلال، ويُقيم بإقامة بلال، والحمد لله، ولا يشذّ على الناس، ويُشوش، ويُسوّي تفرقةً.

قُلْ يَا أَهْلَ الْكِتَابِ هَلْ تَنْقِمُونَ مِنَّا إِلَّا أَنْ آمَنَّا بِاللَّهِ وَمَا أُنْزِلَ إِلَيْنَا وَمَا أُنْزِلَ مِنْ قَبْلُ وَأَنَّ أَكْثَرَكُمْ فَاسِقُونَ [المائدة:59].

الشيخ: بركة.
