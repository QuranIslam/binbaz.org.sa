  5. تفسير قوله: {لَنْ يَسْتَنْكِفَ الْمَسِيحُ أَنْ يَكُونَ عَبْدًا لِلَّهِ وَلَا الْمَلَائِكَةُ الْمُقَرَّبُونَ..}

تفسير قوله: {لَنْ يَسْتَنْكِفَ الْمَسِيحُ أَنْ يَكُونَ عَبْدًا لِلَّهِ وَلَا الْمَلَائِكَةُ الْمُقَرَّبُونَ..}

 
  •  
تحميل المادة

لَنْ يَسْتَنْكِفَ الْمَسِيحُ أَنْ يَكُونَ عَبْدًا لِلَّهِ وَلَا الْمَلَائِكَةُ الْمُقَرَّبُونَ وَمَنْ يَسْتَنْكِفْ عَنْ عِبَادَتِهِ وَيَسْتَكْبِرْ فَسَيَحْشُرُهُمْ إِلَيْهِ جَمِيعًا ۝ فَأَمَّا الَّذِينَ آمَنُوا وَعَمِلُوا الصَّالِحَاتِ فَيُوَفِّيهِمْ أُجُورَهُمْ وَيَزِيدُهُمْ مِنْ فَضْلِهِ وَأَمَّا الَّذِينَ اسْتَنْكَفُوا وَاسْتَكْبَرُوا فَيُعَذِّبُهُمْ عَذَابًا أَلِيمًا وَلَا يَجِدُونَ لَهُمْ مِنْ دُونِ اللَّهِ وَلِيًّا وَلَا نَصِيرًا [النساء:172-173].

قال ابنُ أبي حاتم: حدَّثنا أبي: حدثنا إبراهيم بن موسى: حدثنا هشام، عن ابن جُريج، عن عطاء، عن ابن عباسٍ: قوله: لَنْ يَسْتَنْكِفَ لن يستكبر.

وقال قتادة: لن يحتشم المسيح أن يكون عبدًا لله: وَلَا الْمَلَائِكَةُ الْمُقَرَّبُونَ.

وقد استدلّ بعضُ مَن ذهب إلى تفضيل الملائكة على البشر بهذه الآية، حيث قال: وَلَا الْمَلَائِكَةُ الْمُقَرَّبُونَ، وليس له في ذلك دلالة؛ لأنَّه إنما عطف الملائكة على المسيح؛ لأنَّ الاستنكاف هو الامتناع، والملائكة أقدر على ذلك من المسيح؛ فلهذا قال: وَلَا الْمَلَائِكَةُ الْمُقَرَّبُونَ، ولا يلزم من كونهم أقوى وأقدر على الامتناع أن يكونوا أفضل.

وقيل: إنما ذُكروا لأنَّهم اتّخذوا آلهةً مع الله، كما اتّخذ المسيحُ، فأخبر تعالى أنَّهم عبيدٌ من عباده، وخلقٌ من خلقه، كما قال تعالى: وَقَالُوا اتَّخَذَ الرَّحْمَنُ وَلَدًا سُبْحَانَهُ بَلْ عِبَادٌ مُكْرَمُونَ [الأنبياء:26]؛ ولهذا قال: وَمَنْ يَسْتَنْكِفْ عَنْ عِبَادَتِهِ وَيَسْتَكْبِرْ فَسَيَحْشُرُهُمْ إِلَيْهِ جَمِيعًا أي: فيجمعهم إليه يوم القيامة، ويفصل بينهم بحكمه العدل الذي لا يجور فيه، ولا يحيف.

الشيخ: والمقصود من هذا التَّحذير من الكبر والتَّعاظم الذي قد يُسبب الامتناع من عبادة الله وطاعته، فالمسيح ابن مريم هو عبدالله ورسوله، لن يستنكف أن يكون عبدًا لله كما تظنّ النَّصارى، وأنَّه إلهٌ يُعبد، وقد كذبوا وضلّوا، فهو عبدٌ من عباد الله، وهكذا الملائكة الذين عبدهم المشركون لن يستنكفوا أن يكونوا عبيدًا لله، فلا يجوز أن يُعبدوا مع الله، هم عبيده، فليس لأحدٍ أن يعبدهم: لا المسيح، ولا الملائكة، ولا غيرهم.

الملائكة الذين أعطاهم اللهُ القوةَ، وهم عنده فوق لن يستنكفوا أن يكونوا عبيدًا لله؛ لما وهبهم اللهُ من العلم الذي يُرشدهم إلى الخضوع لله وطاعته؛ ولهذا قال فيهم: بَلْ عِبَادٌ مُكْرَمُونَ ۝ لَا يَسْبِقُونَهُ بِالْقَوْلِ وَهُمْ بِأَمْرِهِ يَعْمَلُونَ ۝ يَعْلَمُ مَا بَيْنَ أَيْدِيهِمْ وَمَا خَلْفَهُمْ وَلَا يَشْفَعُونَ إِلَّا لِمَنِ ارْتَضَى وَهُمْ مِنْ خَشْيَتِهِ مُشْفِقُونَ [الأنبياء:26-28].

فإذا كان المسيحُ وغيره من الأنبياء والملائكة لا يستنكفون، بل يكونون عبيدًا لله، فغيرهم من باب أولى يجب أن يخضعوا لله، وأن يعبدوه وحده، وأن يحذروا الشِّرك به، والتَّكبر عن عبادته.

أمَّا ما يتعلق بتفضيل الملائكة على البشر، والبشر على الملائكة: فالذي عليه أهلُ السُّنة والجماعة أنَّ الصَّالحين من البشر أفضل من الملائكة، صالحو البشر من بني آدم أفضل؛ لأنَّهم ابتُلوا بالتَّكاليف والعبادات، أُعطوا عقولًا، وأُعطوا شهوات، فإذا خضعوا لله وخالفوا شهواتهم وأطاعوا الله صاروا أكمل وأفضل عند ربهم من الملائكة، نعم.

س: ..............؟

ج: يعني: يُزين لهم الشيطان أنَّه إن كنتم ملائكة فأنتم أفضل.

ولهذا قال: فَأَمَّا الَّذِينَ آمَنُوا وَعَمِلُوا الصَّالِحَاتِ فَيُوَفِّيهِمْ أُجُورَهُمْ وَيَزِيدُهُمْ مِنْ فَضْلِهِ أي: فيُعطيهم من الثَّواب على قدر أعمالهم الصَّالحة، ويزيدهم على ذلك من فضله وإحسانه وسعة رحمته وامتنانه.

وقد روى ابنُ مردويه من طريق بقية، عن إسماعيل بن عبدالله الكندي، عن الأعمش، عن سفيان، عن عبدالله مرفوعًا.

الشيخ: هذا مُنقطع؛ فسفيان ما أدرك عبدالله.

مُداخلة: عندنا: عن شقيق، عن عبدالله.

الشيخ: نعم، جيد، نسخة: "عن شقيق"، ويُراجع ابن ماجه أيضًا، حطّ عليه نسخة: "عن شقيق، عن عبدالله"، ويُراجع ابن ماجه.

س: بدل "سفيان": شقيق؟

ج: لا، بعده، بعد: سفيان، شقيق هو: أبو وائل، شقيق بن سلمة، صاحب ابن مسعودٍ، وشيخ سفيان.

قال: قال رسولُ الله ﷺ: فَيُوَفِّيهِمْ أُجُورَهُمْ وَيَزِيدُهُمْ مِنْ فَضْلِهِ قال: أُجُورَهُمْأدخلهم الجنة، وَيَزِيدُهُمْ مِنْ فَضْلِهِ قال: الشَّفاعة فيمَن وجبت له النار ممن صنع إليهم المعروف في دُنياهم. وهذا إسنادٌ لا يثبت.

وإذا رُوِيَ عن ابن مسعودٍ موقوفًا فهو جيد.

الشيخ: على كل حالٍ، محل نظرٍ؛ لأنَّ الأعمش عنعن، وإسماعيل بن عبدالله الكندي انظره في "التقريب".

الطالب: ما ذكره في "التقريب".

الشيخ: انظر "التَّعجيل".

س: عنعنة الأعمش على إطلاقها؟

ج: هو مُدلّس، لكن إذا تُوبع في هذا، أو لم يتبين شيءٌ، فالأصل السَّماع، إلا إذا تبين ما يدلّ على تدليسه، وإلا فالأصل السَّماع، نعم؛ لأنه ثقة، حافظ، والأصل السَّماع.

وَأَمَّا الَّذِينَ اسْتَنْكَفُوا وَاسْتَكْبَرُوا أي: امتنعوا من طاعة الله وعبادته، واستكبروا عن ذلك، فيُعذّبهم عذابًا أليمًا، وَلَا يَجِدُونَ لَهُمْ مِنْ دُونِ اللَّهِ وَلِيًّا وَلَا نَصِيرًا، كقوله: إِنَّ الَّذِينَ يَسْتَكْبِرُونَ عَنْ عِبَادَتِي سَيَدْخُلُونَ جَهَنَّمَ دَاخِرِينَ [غافر:60] أي: صاغرين، حقيرين، ذليلين، كما كانوا مُمتنعين، مُستكبرين.

يَا أَيُّهَا النَّاسُ قَدْ جَاءَكُمْ بُرْهَانٌ مِنْ رَبِّكُمْ وَأَنْزَلْنَا إِلَيْكُمْ نُورًا مُبِينًا ۝ فَأَمَّا الَّذِينَ آمَنُوا بِاللَّهِ وَاعْتَصَمُوا بِهِ فَسَيُدْخِلُهُمْ فِي رَحْمَةٍ مِنْهُ وَفَضْلٍ وَيَهْدِيهِمْ إِلَيْهِ صِرَاطًا مُسْتَقِيمًا [النساء:174-175].

يقول تعالى مُخاطبًا جميع الناس، ومُخبرًا بأنَّه قد جاءهم منه برهانٌ عظيمٌ، وهو الدَّليل القاطع للعذر، والحجّة المزيلة للشُّبهة؛ ولهذا قال: وَأَنْزَلْنَا إِلَيْكُمْ نُورًا مُبِينًا أي: ضياءً واضحًا على الحقِّ. قال ابنُ جريج وغيره: وهو القرآن.

الشيخ: يعمّ القرآن والسُّنة، فالله جعل لعباده نورًا يهتدون به من كتابه وسُنَّة رسوله -عليه الصلاة والسلام-: قَدْ جَاءَكُمْ مِنَ اللَّهِ نُورٌ وَكِتَابٌ مُبِينٌ [المائدة:15]، فالنور يشمل ما جاءت به السُّنة، مع ما جاء به القرآن: إِنَّا أَرْسَلْنَاكَ شَاهِدًا وَمُبَشِّرًا وَنَذِيرًا ۝ وَدَاعِيًا إِلَى اللَّهِ بِإِذْنِهِ وَسِرَاجًا مُنِيرًا [الأحزاب:45-46]، فالله جعل وحيه نورًا يهتدي به الناس، ويُبصرون به مواضع الشُّبهات ومواضع الشَّهوات، حتى يهتدوا إلى السَّبيل، والسُّنة هي التي تُفسّر القرآن، وتُبين ما قد يخفى، الله المستعان، نعم.

فَأَمَّا الَّذِينَ آمَنُوا بِاللَّهِ وَاعْتَصَمُوا بِهِ أي: جمعوا بين مقامي العبادة والتَّوكل على الله في جميع أمورهم.

الشيخ: يعني: إيمان بالله صادق، وتوكّل على الله ، مقامان: مقام الإيمان، ومقام التَّوكل.

وقال ابنُ جريج: آمنوا بالله، واعتصموا بالقرآن. رواه ابنُ جرير.

فَسَيُدْخِلُهُمْ فِي رَحْمَةٍ مِنْهُ وَفَضْلٍ أي: يرحمهم فيُدخلهم الجنة، ويزيدهم ثوابًا ومُضاعفةً ورفعًا في درجاتهم من فضله عليهم وإحسانه إليهم.

وَيَهْدِيهِمْ إِلَيْهِ صِرَاطًا مُسْتَقِيمًا أي: طريقًا واضحًا، قصدًا، قوامًا، لا اعوجاجَ فيه، ولا انحراف، وهذه صفة المؤمنين في الدنيا والآخرة، فهم في الدنيا على منهاج الاستقامة وطريق السَّلامة في جميع الاعتقادات والعمليات، وفي الآخرة على صراط الله المستقيم المفضي إلى روضات الجنان.

وفي حديث الحارث الأعور، عن علي بن أبي طالب ، عن النبي ﷺ أنَّه قال: القرآن صراط الله المستقيم، وحبل الله المتين، وقد تقدّم الحديثُ بتمامه في أول التَّفسير، ولله الحمد والمنَّة.

الشيخ: تقدّم مرفوعًا، وقال المؤلف: أنَّه أشبه أن يكون موقوفًا. وهو حديث عليٍّ: تكون فتنٌ، قيل: فما المخرج منها يا رسول الله؟ قال: كتاب الله، فيه نبأ ما قبلكم، وخبر ما بعدكم، وحكم ما بينكم، وهو الفصل، ليس بالهزل ... إلى آخره، تقدّم في أول التفسير، قال المؤلف -رحمه الله-: والأشبه أن يكون موقوفًا؛ لأنَّ في سنده الحارث، والحارث ضعيف، نعم، الحارث الأعور، نعم.
