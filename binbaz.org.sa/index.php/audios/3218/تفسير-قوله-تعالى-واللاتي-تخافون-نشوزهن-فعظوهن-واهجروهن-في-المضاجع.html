  4. 004- سورة النساء
  5. تفسير قوله تعالى: {وَاللَّاتِي تَخَافُونَ نُشُوزَهُنَّ فَعِظُوهُنَّ وَاهْجُرُوهُنَّ فِي الْمَضَاجِعِ..}

تفسير قوله تعالى: {وَاللَّاتِي تَخَافُونَ نُشُوزَهُنَّ فَعِظُوهُنَّ وَاهْجُرُوهُنَّ فِي الْمَضَاجِعِ..}

 
  •  
تحميل المادة

وَقَوْلُهُ تَعَالَى: وَاللَّاتِي تَخَافُونَ نُشُوزَهُنَّ [النساء:34] أَيْ: وَالنِّسَاءُ اللَّاتِي تَتَخَوَّفُونَ أَنْ يَنْشُزْنَ عَلَى أَزْوَاجِهِنَّ، وَالنُّشُوزُ هُوَ الِارْتِفَاعُ، فَالْمَرْأَةُ النَّاشِزُ هِيَ الْمُرْتَفِعَةُ عَلَى زَوْجِهَا، التَّارِكَةُ لِأَمْرِهِ، الْمُعْرِضَةُ عَنْهُ، الْمُبْغِضَةُ لَهُ، فَمَتَى ظَهَرَ لَهُ مِنْهَا أَمَارَاتُ النُّشُوزِ فَلْيَعِظْهَا وَلْيُخَوِّفْهَا عِقَابَ اللَّهِ فِي عِصْيَانِهِ، فَإِنَّ اللَّهَ قَدْ أَوْجَبَ حَقَّ الزَّوْجِ عَلَيْهَا وَطَاعَتَهُ، وَحَرَّمَ عَلَيْهَا مَعْصِيَتَهُ؛ لِمَا لَهُ عَلَيْهَا مِنَ الْفَضْلِ وَالْإِفْضَالِ، وَقَدْ قَالَ رَسُولُ اللَّهِ ﷺ: لَوْ كُنْتُ آمِرًا أَحَدًا أَنْ يَسْجُدَ لِأَحَدٍ لَأَمَرْتُ الْمَرْأَةَ أَنْ تَسْجُدَ لِزَوْجِهَا مِنْ عِظَمِ حَقِّهِ عَلَيْهَا.

وَرَوَى الْبُخَارِيُّ عَنْ أَبِي هُرَيْرَةَ  قَالَ: قَالَ رَسُولُ اللَّهِ ﷺ: إِذَا دَعَا الرَّجُلُ امْرَأَتَهُ إِلَى فِرَاشِهِ فَأَبَتْ عَلَيْهِ لَعَنَتْهَا الْمَلَائِكَةُ حَتَّى تُصْبِحَ.

وَرَوَاهُ مُسْلِمٌ، وَلَفْظُهُ: إِذَا بَاتَتِ الْمَرْأَةُ هَاجِرَةً فِرَاشَ زَوْجِهَا لَعَنَتْهَا الْمَلَائِكَةُ حَتَّى تُصْبِحَ؛ وَلِهَذَا قَالَ تَعَالَى: وَاللَّاتِي تَخَافُونَ نُشُوزَهُنَّ فَعِظُوهُنَّ.

وَقَوْلُهُ: وَاهْجُرُوهُنَّ فِي الْمَضَاجِعِ [النساء:34].

قَالَ عَلِيُّ ابْنُ أَبِي طَلْحَةَ: عَنِ ابْنِ عباسٍ: الهجر هُوَ أَلَّا يُجَامِعَهَا وَيُضَاجِعَهَا عَلَى فِرَاشِهَا، وَيُوَلِّيَهَا ظَهْرَهُ. وَكَذَا قَالَ غَيْرُ وَاحِدٍ.

وَزَادَ آخَرُونَ مِنْهُمُ السُّدِّيُّ وَالضَّحَّاكُ وَعِكْرِمَةُ وَابْنُ عَبَّاسٍ فِي رِوَايَةٍ: وَلَا يُكَلِّمُهَا مَعَ ذَلِكَ وَلَا يُحدِّثها.

وقال عَلِيُّ ابْنُ أَبِي طَلْحَةَ أَيْضًا: عَنِ ابْنِ عَبَّاسٍ: يَعِظُهَا، فَإِنْ هِيَ قَبِلَتْ وَإِلَّا هَجَرَهَا فِي الْمَضْجَعِ، وَلَا يُكَلِّمهَا مِنْ غَيْرِ أَنْ يَذَرَ نِكَاحَهَا، وَذَلِكَ عَلَيْهَا شَدِيدٌ.

الشيخ: أن يذر، يتركها، هذا محل نظرٍ، الظَّاهر أنَّ هجرَها يشمل هذا وهذا؛ يشمل ترك النِّكاح -ترك الجماع- ويشمل ترك الكلام، وأن يُوليها ظهره، أو يكون في فراشٍ آخر، الهجر يشمل هجرها بالكلام، ويشمل هجرها في الفراش، يكون له فراشٌ آخر، أو يُوليها ظهره، إذا جامعها ما تمَّ الهجرُ، هذا قولٌ ليس بجيدٍ، لكن يكون في البيت، لا يهجر إلا في البيت، كما في حديث معاوية بن حيدة: اهجرها لكن في البيت.

س: يكون في غرفةٍ أخرى، أو يلزم أن يكون في نفس المكان؟

ج: ما هو بلزومٍ، ولو في غرفةٍ أخرى ما يُخالف.

س: حديث السّجود صحيحٌ؟

ج: قد تتبّعتُه سابقًا منذ سنوات، وغالب ظنِّي أنه لا بأس بإسناده، له طرقٌ، وأنا بعيد العهد به، لكني تتبعتُه منذ عدّة سنوات، أُخبركم إن شاء الله في درسٍ آخر، أُخبركم بهذا في درسٍ آخر غدًا أو بعده إن شاء الله، غالب ظني أني قد جمعتُ طرقَه، وتتبعتُ طرقَه.

س: إذا كان في غرفةٍ أخرى ألا يخشى عليها من الفساد؟

ج: لا، ما داما في بيتٍ واحدٍ يُراقبها، وإذا كانا في الحجرة جميعًا وفراشه لحاله كل هذا هجرٌ، النبي عليه السَّلام قال في حديث بهز: ولا تهجر إلا في البيت، إلَّا إذا كان لزوجها عدّة زوجات فلا بأس أن يهجرهنَّ جميعًا في محلٍّ آخر، كما هجر النبيُّ زوجاته وانتقل إلى مشربةٍ خارج البيوت، هجرهنَّ كلّهن في بيوتهن، وانتقل لمشربةٍ خارجيةٍ شهرًا لما أكثرن عليه في النَّفقة.

وَقَالَ مُجَاهِدٌ وَالشَّعْبِيُّ وَإِبْرَاهِيمُ وَمُحَمَّدُ بْنُ كَعْبٍ وَمِقْسَمٌ وَقَتَادَةُ: الْهَجْرُ هُوَ أَنْ لَا يُضَاجِعَهَا.

الشيخ: لأنَّ الله قال: وَاهْجُرُوهُنَّ فِي الْمَضَاجِعِ [النساء:34] يعني: محلّ النوم، المضجع، يكون له فراشٌ، ولها فراشٌ.

والحاصل من هذا كما يأتي أنَّ الرجلَ إذا رأى من زوجته أمارات النُّشوز يعظها ويُذكّرها ويُحاسِب نفسَه هو؛ لأنَّه قد يكون نشوزُها بأسبابه؛ بأسباب سُوء سيرته، وأسباب خلقه السَّيئ، فليُحاسِب نفسَه أولًا وينظر، فإذا كان قد قام بحقِّها فحينئذٍ وجب عليها حقُّه، لكن بعض الأزواج سيئ الخلق، سيئ السيرة، يسهر في الليل ويُخليها لحالها وحدها في الليل، وإذا جاءها فهو تعبان أو سكران، وبعض الأحيان يكون كلامُه معها سيئًا، وقيامه بحقِّها قليل، هذه معذورة؛ لأنَّه قصَّر في حقِّها، فالواجب عليه أن يقوم بحقِّها هو، فلا تُسمَّى حينئذٍ: ناشزًا، هو النَّاشز، هو الظالم.

أمَّا إذا قام بحقِّها وأدَّى حقَّها، ولكن هي أبت؛ هذا هو النُّشوز، هي التي تستحقّ حينئذٍ أن يعظها ويهجرها ويضربها على حسب الحال، لكن في هذا الوقت لا نُحصي ولا نعدّ مَن تشكو زوجَها من جهة الظلم وعدم المبالاة بحقِّها؛ يسهر مع جماعته -فسقة وغير فسقة- فإذا جاء آخر الليل قد هلك، ميتًا، وإذا خاطبها خاطبها بالسَّبِّ واللَّعن والضَّرب، فأيُّ زوجٍ هذا؟!

وقيامه بما يلزم من النَّفقة كذلك، قد تمر عليه الأيام والليالي والشّهور لا يعرف بيتًا، ولا يُبالي ببيته، بعض الناس جديرٌ بالضَّرب هو والعقوبة الشَّديدة، فإذا أدَّى حقَّها واستقام في حقِّها فله الحقُّ عليها في أن يهجرها ويعظها ويضربها؛ لأنَّها هي المقصّرة.

وَقَدْ قَالَ أَبُو دَاوُدَ: حَدَّثَنَا مُوسَى بْنُ إِسْمَاعِيلَ: حَدَّثَنَا حَمَّادُ بْنُ سَلَمَةَ، عَنْ عَلِيِّ بْنِ زَيْدٍ، عَنْ أَبِي حرَّةَ الرَّقَاشِيِّ.

.............

الطالب: أبو حرّة -بضمِّ أوله وتشديد الرَّاء- البصري، اسمه: واصل بن عبدالرحمن.

أبو حرّة الرّقاشي اسمه: حنيفة.

الشيخ: بالحاء.

عَنْ عَمِّهِ: أَنَّ النَّبِيَّ ﷺ قَالَ: فَإِنْ خِفْتُمْ نُشُوزَهُنَّ فَاهْجُرُوهُنَّ فِي الْمَضَاجِعِ، قَالَ حَمَّادٌ: يَعْنِي النِّكَاحَ.

وَفِي السُّنَنِ وَالْمُسْنَدِ عَنْ مُعَاوِيَةَ بْنِ حَيْدَةَ الْقُشَيْرِيِّ أَنَّهُ قَالَ: يَا رَسُولَ اللَّهِ، مَا حَقُّ امرأة أحدنا عليه؟ قَالَ: أَنْ تُطْعِمَهَا إِذَا طَعِمْتَ، وَتَكْسُوهَا إِذَا اكْتَسَيْتَ، وَلَا تَضْرِبَ الْوَجْهَ، وَلَا تُقَبِّحَ، وَلَا تَهْجُرْ إِلَّا فِي الْبَيْتِ.

وَقَوْلُهُ: وَاضْرِبُوهُنَّ، أَيْ إذا لم يرتد عن بِالْمَوْعِظَةِ وَلَا بِالْهِجْرَانِ، فَلَكُمْ أَنْ تَضْرِبُوهُنَّ ضَرْبًا غَيْرَ مُبَرِّحٍ، كَمَا ثَبَتَ فِي صَحِيحِ مُسْلِمٍ .......

وَقَالَ سُفْيَانُ بْنُ عُيَيْنَةَ: عَنِ الزُّهْرِيِّ، عَنْ عَبْدِاللَّهِ بْنِ عَبْدِاللَّهِ بْنِ عُمَرَ، عَنْ إِيَاسِ بْنِ عَبْدِالله ابن أبي ذباب قَالَ: قَالَ النَّبِيُّ ﷺ: لا تضربوا إماءَ الله، فجاء عمرُ  إلى رسول الله ﷺ فقال: ذئرت النِّساءُ على أزواجهنَّ. فرخَّص رسولُ الله ﷺ في ضَرْبِهِنَّ، فَأَطَافَ بِآلِ رَسُولِ اللَّهِ ﷺ نِسَاءٌ كَثِيرٌ يَشْكُونَ أَزْوَاجَهُنَّ، فَقَالَ رَسُولُ اللَّهِ ﷺ: لَقَدْ أَطَافَ بِآلِ مُحَمَّدٍ نِسَاءٌ كَثِيرٌ يَشْكُونَ أَزْوَاجَهُنَّ، لَيْسَ أُولَئِكَ بِخِيَارِكُمْ. رَوَاهُ أَبُو دَاوُدَ وَالنَّسَائِيُّ وَابْنُ مَاجَهْ.

الشيخ: يعني ليسوا بخِياركم، يعني: الضَّرابين للنِّساء، يعني: عالجوا بغير الضَّرب مهما أمكن.

ومُقتضى الأدلة أنه ينبغي للزوج أن يُعالج مهما أمكن بغير ضربها؛ لأنَّ الضَّربَ قد يُؤثر بغضًا شديدًا وفرقةً واختلافًا، فمهما أمكن أن يُعالج بالهجر والوعظ وإرضائها بالشَّيء الذي يستطيعه من المال أو نحو ذلك، هذا هو الأولى بالرجل؛ العلاج بالطُّرق الأخرى غير الضَّرب مهما أمكن، يكون الضربُ هو آخر الطب، وضربًا غير مُبرح، هذا هو الأولى بالرجل، وهو الأقرب إلى الوئام وصلاح الحال.

س: ............؟

ج: يبغضها، يضربها إذا أخلَّت بواجبٍ، أما البغضاء ما هي بيده، البغضاء بيد الله، لكن في القلوب هذه، لكن إذا قصَّرت في حقِّها ولم ينفع الوعظُ، ولم ينفع الهجرُ جاز الضَّرب، وإن اختلعت فلا بأس -طلبت الخلع- خلعها؛ ولهذا لما اشتدَّت البغضاءُ بين جميلة بنت عبدالله بن أُبي وبين ثابت بن قيس قالت: يا رسول الله، ما أُطيقه، ولا أعتب عليه من خلقٍ ولا دينٍ. ما أعيب عليه ..... في خلقه ولا دينه، لكن بغضاء، فقال: أتردّين عليه حديقته؟ قالت: نعم. فقال له: اقبل الحديقةَ وطلِّقها تطليقة. رواه البخاريُّ رحمه الله، فإذا اشتدّت الحالُ وساءت الحالُ فإنه يُخالعها.

س: ضربها بيده أو بعصا ونحوها؟

ج: باليد أو بالعصا، نعم، لكن يكون في محلٍّ ليس في محلِّ خطرٍ، ولا في وجهٍ أيضًا.

س: الحديث السَّابق في "صحيح مسلم" في قوله: وأن لا يُوطئن فرشَكم أحدًا تكرهونه، فإن فعلن فاضربوهنَّ ضربًا غير مُبَرِّحٍ؟

ج: رواه مسلم في "الصحيح"، وأيش فيه؟

س: ما المقصود بـ: وأن لا يُوطئن فرشَكم أحدًا ..؟

ج: يعني يأذنون في البيوت لمن لا يرضاه الزوج، يقول: لا يدخل فلانٌ ولا فلانة. وتساهلت وتُدخله، من جيرانٍ أو من غيرهم، إذا قال: لا يجيئنا فلان ولا فلانة. لا تسمح لهما.

وَقَالَ الْإِمَامُ أَحْمَدُ: حَدَّثَنَا سُلَيْمَانُ بْنُ دَاوُدَ -يَعْنِي أَبَا دَاوُدَ الطَّيَالِسِيَّ- حَدَّثَنَا أَبُو عَوَانَةَ، عَنْ دَاوُدَ الْأَوَدِيِّ، عَنْ عَبْدِالرَّحْمَنِ الْمُسْلِيِّ، عَنِ الْأَشْعَثِ بْنِ قَيْسٍ قَالَ: ضفتُ عمر ، فتناول امرأته فضربها، فقال: يا أشعث، احفظ عني ثلاثًا حفظتُها عَنْ رَسُولِ اللَّهِ ﷺ: لَا تَسْأَلِ الرَّجُلَ فِيمَ ضَرَبَ امْرَأَتَهُ، وَلَا تنم إلا على وترٍ، ونسيتُ الثَّالِثَةَ.

وَكَذَا رَوَاهُ أَبُو دَاوُدَ وَالنَّسَائِيُّ وَابْنُ مَاجَهْ مِنْ حَدِيثِ عَبْدِالرَّحْمَنِ بْنِ مَهْدِيٍّ، عَنْ أَبِي عَوَانَةَ، عَنْ دَاوُدَ الْأَوَدِيِّ بِهِ.

الشيخ: على كل حالٍ عند هؤلاء الأربعة: أحمد وأبي داود والنَّسائي وابن ماجه ..... أبي عبدالرحمن السّلمي فيما يتعلَّق بالنِّكاح.

وقوله تعالى: فَإِنْ أَطَعْنَكُمْ فَلَا تَبْغُوا عَلَيْهِنَّ سَبِيلًا [النساء:34] أَيْ: إذا أطاعت المرأةُ زوجَها في جميع ما يُريده مِنْهَا مِمَّا أَبَاحَهُ اللَّهُ لَهُ مِنْهَا فَلَا سَبِيلَ لَهُ عَلَيْهَا بَعْدَ ذَلِكَ، وَلَيْسَ لَهُ ضَرْبُهَا وَلَا هِجْرَانُهَا.

وَقَوْلُهُ: إِنَّ اللَّهَ كَانَ عَلِيًّا كَبِيرًا [النساء:34] تَهْدِيدٌ لِلرِّجَالِ إِذَا بَغَوْا عَلَى النِّسَاءِ مِنْ غَيْرِ سَبَبٍ، فَإِنَّ اللَّهَ الْعَلِيَّ الكبيرَ وليهنَّ، وهو ينتقم ممن ظلمهنَّ وبغى عليهنَّ.

الشيخ: وهذه وصيةٌ من ربِّنا أن يحذر الزوج ثم المرأة، مثلما قال ﷺ: استوصوا بالنِّساء خيرًا، فإنهنَّ عوانٌ عندكم يعني: أسيرات، قلَّ مَن تستطيع منهنَّ أن تنتصر من الزوج، الغالب عليهنَّ الضَّعف أمام الزوج، والزوج يكون عنده في الغالب القُدرة على الإيذاء والضَّرب.

فالواجب تقوى الله، والحذر من الظلم، والحرص على الخلق الجميل الكريم الطَّيب، والصِّفات الحميدة، وأن تُعطى حقَّها، فإذا دعت الحاجةُ القصوى إلى الضَّرب فهذا هو محل التَّأديب.

مُداخلة: الشيخ فهد ..... أتى بتخريج حديث: "لا يُسأل الرجلُ فيمَ ضرب امرأتَه" من مسند الإمام أحمد.

الشيخ: طيب، معاك؟

الطالب: إيه، نعم.

الشيخ: تقرأه.

الطالب: قال الإمام أحمد في "مسنده": حدَّثنا سليمان بن داود -يعني أبا داود الطيالسي- قال: حدَّثنا أبو عوانة، عن داود الأودي، عن عبدالرحمن المسلي، عن الأشعث بن قيسٍ قال: ضفتُ عمر ، فتناول امرأته فضربها، وقال: يا أشعث، احفظ عني ثلاثًا حفظتُهنَّ عن رسول الله ﷺ: لا تسأل الرجل فيمَ ضرب امرأته، ولا تنم إلا على وترٍ، ونسيتُ الثالثة.

ورواه أبو داود، والنَّسائي في "الكبرى"، وابن ماجه، والطَّحاوي في "مشكل الآثار"، والبيهقي، كلّهم من طريق أبي عوانة به، ووقع عندهم إلا البيهقي: "لا يسأل"، بالياء التَّحتانية على البناء للمجهول، وهذا سندٌ رجاله كلّهم ثقات إلا عبدالرحمن المسلي، وهو بضم الميم وسكون السين المهملة، قال أبو الفتح الأزدي: فيه نظر. وقال الذَّهبي في "الميزان": لا يُعرف إلا في حديثه عن الأشعث، عن عمر : لا تسأل الرجل فيمَ ضرب امرأته. تفرَّد عنه داود بن عبدالله الأودي، وقال الحافظ في "التقريب": مقبول.

وأما داود الأودي فهو داود بن عبدالله الأودي، كما عند أبي داود والنَّسائي وابن ماجه، وهو ثقة. قاله أحمد وابن معين وأبو داود وغيرهم، وشذَّ ابنُ حزمٍ فقال ..... مجهول. وقال الحافظ في "التقريب": ثقة. وهو غير داود بن يزيد الأودي عمِّ ابن إدريس، فإنَّه ضعيفٌ، وأمَّا ابن عبدالله فثقة.

الشيخ: داود بن عبدالله يعني.

الطالب: تنبيهٌ: وقع في "تحفة الأشراف": عبدالرحمن ابن أبي ليلى، بدل: عبدالرحمن المسلي، قال الحافظُ في "النُّكَت": كذا بخط المزي، والصَّواب: عبدالرحمن المسلي.

الشيخ: وبهذا يكون الحديثُ ليس بذاك؛ لأنَّ المقبولَ هو الذي يكون له شواهد، المقبول إذا جاء للحديث شواهد، وهذا له شواهد، لكن ليس فيها ذكر الثالثة، ومن شواهد هذا الحديث وصية النبي لأبي هريرة وأبي الدَّرداء بركعتي الضُّحى، والوتر قبل النوم، وصيام ثلاثة أيام من كل شهرٍ، هذا شاهدٌ لقوله: لا تنم إلا على وترٍ، فهذا هو الأفضل لمن يخشى ألا يقوم من آخر الليل، أمَّا مَن يطمع أن يقوم آخر الليل ولا يشقّ عليه إذا تيسر فهو أفضل، آخر الليل أفضل، لكن مَن خاف ألا يقوم فالسُّنة ألا ينام إلا على وترٍ.

وأمَّا النهي عن سؤال ضرب امرأته فهو من حيث المعنى واضح؛ لأنها قد تكون أشياء سرية، فكونك تسأله تدخل فيما لا ينبغي، قد يكون ضربها في امتناعها من الجماع، قد يكون في أشياء سرية، وفي سؤالها عن أسباب الضَّرب إحراجٌ، اللهم إلا عند الضَّرورة والخصومة: كعند القاضي، وعند الحكمين، قد تدعو الحاجة إلى هذا.

فهذا الحديث من جهة سؤال الرجل عن امرأته إذا لم يكن له شواهد فهذا السَّند ضعيفٌ؛ لأنَّ عبدالرحمن هذا ليس ممن يُحتجّ به حتى تكون له شواهد، ولكن إذا دعت الحاجةُ إلى الخصومة في ضربه إياها، وعند سؤال الحكمين؛ لأنَّ الحاجة تدعو إلى ذلك فلا بأس.

س: ...............؟

ج: كذلك هذا من نكارته، إلَّا أن يُحمل هذا على أنَّ هذا قبل الحجاب، لكن هذا في عهد عمر مُتأخّر، هذا فيه نكارة من هذه الحيثية؛ لأنَّ هذا مُتأخر، هذا لا شكَّ إلا أن يُحمل على أنه تناولها يعني: سمع صوتها، وسمع صياحها، ولا يلزم منه الرؤية.

ج: المسلي، ابن ماجه فيه أغلاط، أقول: المسلي، المسلي، الميم الثانية زائدة.
